        function inicializar(url){
            http=getXMLHTTPRequest();
            http.onreadystatechange = procesarEvento;
            http.open("GET", url, true);
            http.send(null);
            tiempo=setTimeout("finDeEspera()",3000);
        }

            var detalles = document.getElementById("detalles");
